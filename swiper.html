Swiper 輪播範例

Swiper 官網

範例程式碼


            <!-- Swiper CSS 和 JS -->
            <link href="https://unpkg.com/swiper/swiper-bundle.min.css" rel="stylesheet">
            <script src="https://unpkg.com/swiper/swiper-bundle.min.js"></script>

            <!-- 初始化 Swiper -->
            <script>
                var swiper = new Swiper('.swiper-container', {
                    loop: true,
                    slidesPerView: 1,
                    autoplay: {
                        delay: 2500,
                    },
                });
            </script>
