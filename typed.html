Typed.js 官網

範例程式碼


            <!-- Typed.js JS -->
            <script src="https://cdn.jsdelivr.net/npm/typed.js@2.0.12"></script>

            <!-- 初始化 Typed.js -->
            <script>
                var options = {
                    strings: ["Hello, World!", "Welcome to Typed.js!"],
                    typeSpeed: 50
                };
                var typed = new Typed(".typed", options);
            </script>
